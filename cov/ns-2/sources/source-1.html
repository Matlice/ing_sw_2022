796     private Category XMLCreateChildCategories(Category root, Deque<AbstractMap.SimpleEntry<XMLImport.CategoryXML, Category>> categoryStack) throws DuplicateCategoryException, InvalidCategoryException, InvalidFieldException, DuplicateFieldException {
797         while (categoryStack.size() != 0) {
798             var toInsert = categoryStack.pop();
799             Category father = toInsert.getValue();
800 
801             //era un while ma viene eseguito solo una volta
802             NodeCategory r = XMLBuildChildCategory(root, categoryStack, toInsert, father);
803 
804             // aggiorna i padri nello stack, serve perchè l'istanza potrebbe essere cambiata
805             categoryStack = getUpdatedCategoryStack(categoryStack, father, r);
806 
807             if (father == root) root = r;
808         }
809         return root;
810     }

819             } else {
820                 newCategoryStack.add(e);
821             }
822         }
823         categoryStack = newCategoryStack;
824         return categoryStack;
825     }
